namespace = giga_orbital_arcology_origin

### Game Start
country_event = {
	id = giga_orbital_arcology_origin.001
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_origin = origin_orbital_arcology
		is_ai = no
	}

	immediate = {
		capital_scope = {
			solar_system = {
				if = { 
					limit = { prev = { planet_size < 13 } }
					spawn_megastructure = { type = "orbital_arcology_1_s" planet = prev }
				}
				else_if = { 
					limit = { prev = { planet_size >= 13 planet_size <= 21 } }
					spawn_megastructure = { type = "orbital_arcology_1_m" planet = prev }
				}
				else_if = {
					limit = { prev = { planet_size > 21 } } 
					spawn_megastructure = { type = "orbital_arcology_1_l" planet = prev }
				}
			}
		}
	}
}
