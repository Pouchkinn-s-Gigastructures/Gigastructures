							exists = solar_system

								urmazin_system = yes
							}

							which = ehof_phase

							has_global_flag = ehof_traders_spawned
						}
					}
				}
				set_timed_country_flag = {
					flag = ehof_recently_visited_trader
					days = @ehof_trader_visiting_days
				}
				random_system = {
					limit = {
						urmazin_capital = yes
					}

				}
			}

				country_event = {
					id = ehof_systems.000
				}

		ehof_mega_system = yes

			id = ehof_megastructure.501
		}
