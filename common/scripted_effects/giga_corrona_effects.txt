increase_solak_resources = {
	event_target:flusion_primitives_country = {
		if = {
			limit = {
				check_variable = {
					which = "solak_resources"
					value < 10
				}
			}
			change_variable = {
				which = "solak_resources"
				value = 1
			}
		}
	}
}

decrease_solak_resources = {
	event_target:flusion_primitives_country = {
		if = {
			limit = {
				check_variable = {
					which = "solak_resources"
					value > 0
				}
			}
			subtract_variable = {
				which = "solak_resources"
				value = 1
			}
		}
	}
}

increase_solak_partisans = {
	if = {
		limit = { event_target:flusion_primitives_country = { check_variable = { which = solak_res value = 0 } } }
		event_target:flusion_primitives_country = {
			set_variable = { which = solak_res value = 1 }
			remove_modifier = solak_partisan_penalties
			add_modifier = { modifier = solak_partisan_penalties multiplier = solak_res }
		}
		every_country = {
			limit = { ehof_default_country = yes }
			add_modifier = {
				modifier = solak_partisans_1
				days = -1
			}
		}
