ehof_generate_pgate = {
	random_system = {
		limit = {
			NOR = {
				has_star_flag = empire_cluster
				has_star_flag = ehof_has_pgate
				closest_system = {
					has_megastructure = ehof_pgate_unknown
					distance = {
						source = prev
						max_distance <= 200
					}
				}
			}
		}
		set_star_flag = ehof_has_pgate
		spawn_megastructure = {
			type = "ehof_pgate_unknown"
			orbit_angle = 110
			orbit_distance = 35
		}
	}
}

ehof_move_fleet_to_ehof = {

    log = "move fleet to EHOF"

    # save ehof system as event target
    ehof_mega_system = yes

    event_target:ehof_megastructure_system = {
        # Move every fleet in system to wormhole
		every_ship_in_system = {
			limit = {
				fleet = {
					can_use_ehof = yes
					is_mobile = yes
					exists = owner
					owner = { is_same_value = root.owner }
				}
				has_component = SUBSPACE_DRIVE
			}
			set_ship_flag = ehof_eager_tp@root.owner
			fleet = {
				set_fleet_flag = eager_ehof@root.owner
			}
		}
        every_fleet_in_system = {
        	limit = {
        		can_use_ehof = yes
				is_mobile = yes
        		exists = owner
        		owner = { is_same_value = root.owner }
				NOT = { has_fleet_flag = eager_ehof@root.owner}
        	}
        	clear_orders = yes
        	set_fleet_flag = ehof_ready_for_travel@root.owner
        	auto_move_to_planet = {
        		target = star
        		clear_auto_move_on_arrival = yes
        	}
        }
    }
}

ehof_recount_ships = {

     # save ehof system as event target
     ehof_mega_system = yes

     owner = { set_variable = { which = ehof_pre_flight_science value = 0 } }
     owner = { set_variable = { which = ehof_pre_flight_constructor value = 0 } }
     owner = { set_variable = { which = ehof_pre_flight_military value = 0 } }
     owner = { set_variable = { which = ehof_pre_flight_colossus value = 0 } }
     owner = { set_variable = { which = ehof_pre_flight_juggernaut value = 0 } }

     event_target:ehof_megastructure_system = {
         # Count every fleet in system to wormhole

         log = "recount ships"

         every_ship_in_system = {
         	limit = {
				
         		can_use_ehof = yes
			# OR = {
			# 	NOT = {
			# 		exists = fleet
			# 	}
				fleet = {
					is_mobile = yes
				}
			# }	
				 
         		exists = owner
         		owner = { is_same_value = root.owner }
         	}

			if = { limit = { ehof_is_science_ship = yes }						owner = { change_variable = { which = ehof_pre_flight_science value = 1 } } }
            if = { limit = { ehof_is_construction_ship = yes }					owner = { change_variable = { which = ehof_pre_flight_constructor value = 1 } } }
            if = { limit = { is_ship_class = shipclass_military }				owner = { change_variable = { which = ehof_pre_flight_military value = 1 } } }
            if = { limit = { is_ship_class = shipclass_transport }				owner = { change_variable = { which = ehof_pre_flight_military value = 1 } } }
            if = { limit = { is_ship_class = shipclass_military_special }		owner = { change_variable = { which = ehof_pre_flight_colossus value = 1 } } }
            if = { limit = { is_ship_size = juggernaut }						owner = { change_variable = { which = ehof_pre_flight_juggernaut value = 1 } } }

         }
     }
}

#These 3 effects can be run in a loop to safely ensure a system is eventually reconnected back to its hyperlane type, YOU HAVE TO LIMIT IT TO ONLY TRY AND RUN FROM THE CORRECT HYPERLANE TYPE THOUGH (Either normal eligible, ehof eligible, or core eligible)
giga_reconnect_system_normal = {
	if = {
		limit = {
			NOT = {
				distance = {
					source = event_target:hyperlane_check_system
					min_jumps = 1 #its 0 jumps since it has no connection
					max_jumps = 999
					use_bypasses = no
				}
			}
		}
		set_spawn_system_batch = begin #This has to be done to force the cache to regenerate and so the effect works when rerun
		random_list = {
			1 = {
				while = { count = 1 giga_generate_hyperlane = yes }
			}
			1 = {
				while = { count = 2 giga_generate_hyperlane = yes }
			}
			1 = {
				while = { count = 3 giga_generate_hyperlane = yes }
			}
		}
		set_spawn_system_batch = end 
	}
}
