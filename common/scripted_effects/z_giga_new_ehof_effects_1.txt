ehof_giga_new_initialize_system_2 = {
	no_scope = {
		spawn_system = {
			min_orientation_angle = $ehof_init_min_angle$
			max_orientation_angle = $ehof_init_max_angle$
			min_distance >= $min_distance$
			max_distance <= $max_distance$
			hyperlane = no
			initializer = "ehof_origin_cohesive_cluster"
			effect = {
				save_event_target_as = giga_last_ehof_spawn
			}
		}
	}
}

		limit = {
			urmazin_system = no

			}

		}
	}

	star = {
		trigger_megastructure_icon = yes
	}
	if = {
		limit = {
			has_global_flag = giga_dyson_scaling
		}
		if = {
			limit = {
				star = {
					is_planet_class = pc_b_star
				}
			}
			spawn_megastructure = {
				type = "dyson_sphere_ruined_b_star"
				owner = from
			}
		}
		else_if = {
			limit = {
				star = {
					is_planet_class = pc_m_giant_star
				}
			}
			spawn_megastructure = {

ehof_giga_new_create_rl_digsite = {
	#Standardized Reality Code looks like this VVV
	#digsite = code1_digsite 
	#digsite_flag = code_world1
	#digsite_completion_flag = ehof_code_1_complete
	if = {
		limit = {
			NOT = {
				exists = archaeological_site
			}
		}
		create_archaeological_site = $digsite$
		if = {
			limit = { 
				exists = event_target:rl_digsite_finder		
				event_target:rl_digsite_finder = {
					NAND = {
						has_country_flag = $digsite_flag$@prev
						prev = {
							has_planet_flag = $digsite_flag$@prev
						}
					}
				}
			}
			event_target:rl_digsite_finder = {
				set_country_flag = $digsite_flag$@prev
				prev = {
					set_planet_flag = $digsite_flag$@prev
				}
			}
		}
	}
	else_if = {
		limit = {
			archaeological_site = {
				not = { is_site_type = $digsite$ } 
			}
		}
		destroy_archaeological_site = archaeological_site
		create_archaeological_site = $digsite$
	}
	else = {
		if = {
			limit = {
				exists = event_target:rl_digsite_finder
			}
			if = {
				limit = { 
					event_target:rl_digsite_finder = {
						has_country_flag = $digsite_flag$@prev
					}
				}
				event_target:rl_digsite_finder = {
					prev = {
						set_planet_flag = $digsite_flag$@prev
					}
				}
			}
			else_if = {
				limit = { 
					event_target:rl_digsite_finder = {
						prev = {
							has_planet_flag = $digsite_flag$@prev
						}
					}
				}
				event_target:rl_digsite_finder = {
					set_country_flag = $digsite_flag$@prev
				}
			}
			else_if = {
				limit = { 
					NOR = {
						any_country = {
							has_country_flag = $digsite_flag$@prev
						}
						any_country = {
							prev = {
								has_planet_flag = $digsite_flag$@prev
							}
						}
						event_target:rl_digsite_finder = {
							has_country_flag = $digsite_completion_flag$
						}
					}
				}
				event_target:rl_digsite_finder = {
					set_planet_flag = $digsite_flag$@prev
					prev = {
						set_country_flag =  $digsite_flag$@prev
					}
				}
			}
		}
	}
}
