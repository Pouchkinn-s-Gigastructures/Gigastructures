giga_penrose_system = {
	class = "sc_black_hole"
	name = "The Eternal Arc"
	usage = custom_empire
    flags = { empire_home_system habitat_system }

	init_effect = {
		set_star_flag = sol_sector ### to stop New Frontiers from overriding 'planet' classes
		every_neighbor_system = {
			set_star_flag = empire_cluster
			every_neighbor_system = {
				set_star_flag = empire_cluster
			}
		}

		spawn_megastructure = { type = relic_penrose_sphere_1 }
	}

	planet = {
		class = pc_black_hole
		orbit_angle = 0
		orbit_distance = 0
		size = 15
		flags = { ignore_startup_effect starting_deposit }

		init_effect = {
			set_name = "The Elder Void"
			clear_deposits = yes
			prevent_anomaly = yes
		}
	}

	change_orbit = 45

	### gigastructures / interstellar ringworld section - block 1
	planet = {
		class = pc_giga_penrose_habitable
		orbit_angle = 0
		orbit_distance = 0
		size = 24
		has_ring = yes

		home_planet = yes

		flags = { ignore_startup_effect megastructure }
		init_effect = {
			set_name = "Penrose Ringworld Section A"
			set_planet_entity = {
				entity = "penrose_sphere_ringworld_01_entity"
				graphical_culture = humanoid_01
			}

			save_event_target_as = penrose_section
			trigger_megastructure_icon = yes

			set_planet_flag = megastructure			# Vanilla uses to check if a megastrucure can be built here
			set_planet_flag = has_megastructure		# Vanilla uses to check if an orbital station can be built here
			set_planet_flag = colony_event			# Vanilla uses to prevent unwanted events on planets
			set_planet_flag = giga_penrose
		}
	}
	### gigastructures / interstellar ringworld section - block 2
	planet = {
		class = pc_giga_penrose_habitable
		orbit_angle = 180
		orbit_distance = 0
		size = 24
		has_ring = yes

		flags = { ignore_startup_effect megastructure }
		init_effect = {
			set_name = "Penrose Ringworld Section B"
			set_planet_entity = {
				entity = "penrose_sphere_ringworld_01_entity"
				graphical_culture = humanoid_01
			}

			clear_blockers = yes
			prevent_anomaly = yes
			save_event_target_as = penrose_section
			trigger_megastructure_icon = yes

			set_planet_flag = megastructure			# Vanilla uses to check if a megastrucure can be built here
			set_planet_flag = has_megastructure		# Vanilla uses to check if an orbital station can be built here
			set_planet_flag = colony_event			# Vanilla uses to prevent unwanted events on planets
			set_planet_flag = giga_penrose
		}
	}

	### neighbouring systems : guarantee an A-class star
	neighbor_system = {
		hyperlane_jumps = { min = 1 max = 3 }
		initializer = "giga_sc_a"
	}
}
