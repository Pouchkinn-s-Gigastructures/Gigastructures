d_giga_elysium_buildings = {
	is_for_colonizable = yes
	icon = d_city
	category = deposit_cat_energy

	drop_weight = { weight = 0 }
	potential = { always = no }

	triggered_planet_modifier = {
		potential = { always = yes }
		planet_max_buildings_add = 1

		multiplier = value:giga_elysium_buildings
	}

	triggered_planet_modifier = {
		potential = { 
			exists = owner
			owner = {
				has_origin = origin_elysium
			}
		}
		planet_buildings_hab_capital_alloys_upkeep_mult = -0.8
	}
}

d_giga_elysium_slums = {
	time = 100

	is_for_colonizable = yes
	icon = d_slums
	category = deposit_cat_blockers

	potential = {
		always = no
	}

	planet_modifier = {
		planet_housing_add = 10
		planet_crime_add = 10
		planet_stability_add = -5
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			or = {
				owner = { is_machine_empire = no }
				any_owned_species = {
					is_robotic_species = no
				}
			}
		}
		pop_growth_speed = 0.05
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		job_criminal_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		job_deviant_drone_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		job_corrupt_drone_add = 1
	}

	resources = {
		category = deposit_blockers
		cost = {
			energy = 300
		}
	}

	can_be_cleared = {
		exists = owner
		or = {
			owner = { is_ai = no }
			and = { 
				free_housing > 10
				planet_stability < 75
			}
			planet_stability < 50
		}
	}
}
