# the custom menu scopes to country using it IF they have nothing selected
# selected stuff like planet or fleet needs to find the owner
defined_text = {
	name = has_orbital_artificial_eco
	text = {
		trigger = {
			if = {
				limit = {
					is_scope_type = country
				}
				OR = {
					has_megastructure = orbital_artificial_eco_3
				}
			}
			else = {
				owner = {
					OR = {
						has_megastructure = orbital_artificial_eco_3
					}
				}
			}
		}
		localization_key = giga_orbital_artificial_eco_yes
	}
	text = {
		trigger = {
			if = {
				limit = {
					is_scope_type = country
				}
				NOT = {
					has_megastructure = orbital_artificial_eco_3
				}
			}
			else = {
				owner = {
					NOT = {
						has_megastructure = orbital_artificial_eco_3
					}
				}
			}
		}

}
