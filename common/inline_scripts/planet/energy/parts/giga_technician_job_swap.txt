
# technicians
triggered_planet_modifier = {
    potential = {
        exists = owner
        owner = {
            is_gestalt = no
            is_fallen_empire_spiritualist = no
        }
        $condition$
    }

    job_technician_add = $jobs$
    $multiplier_string$
}
inline_script = {
    script = generic_parts/giga_toggled_code
    code = "
        triggered_desc = {
            trigger = {
                exists = owner
                owner = {
                    is_gestalt = no
                    is_fallen_empire_spiritualist = no
                }
                $condition$
            }
            text = job_technician_effect_desc
        }
    "
    toggle = $description$
}
