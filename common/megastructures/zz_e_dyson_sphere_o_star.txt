#Dyson Sphere
dyson_sphere_5_o_star = {
	entity = "dyson_sphere_phase_05_o_star_entity"
	construction_entity = "dyson_sphere_part_4_o_star_entity"
	portrait = "GFX_megastructure_dyson_sphere_background"
	upgrade_from = { dyson_sphere_4_o_star }
	prerequisites = { "tech_dyson_sphere" }
	ai_weight = { factor = value:giga_ai_base_continue } # If we can build it, we will build it

	build_time = @dyson_p5_buildtime_o_star
	resources = {
		category = giga_megastructures
		cost = { alloys = @dyson_p5_cost_alloys_o_star unity = @dyson_cost_unity_o_star }
		cost = { alloys = -1 multiplier = value:giga_ai_savings_cost|CATEGORY|important|RESOURCE|alloys|AMOUNT|@dyson_p5_cost_alloys_o_star| }
		cost = { unity = -1 multiplier = value:giga_ai_savings_cost|CATEGORY|important|RESOURCE|unity|AMOUNT|@dyson_cost_unity_o_star| }
		produces = { energy = @dyson_p5_produces_energy_o_star }
	}

	on_build_start = {
		giga_ai_savings_withdraw = { CATEGORY = important RESOURCE = alloys AMOUNT = @dyson_p5_cost_alloys_o_star }
		giga_ai_savings_withdraw = { CATEGORY = important RESOURCE = unity  AMOUNT = @dyson_cost_unity_o_star }
	}

	on_build_complete = {
		if = {
			limit = {
				count_system_planet = {
					limit = { is_star = yes }
					count = 1
				}
			}
			set_star_class = sc_dyson_star
			every_system_planet = {
				limit = { is_star = yes }
				change_pc = pc_giga_o_star
			}
			every_system_ambient_object = {
				limit = { has_ambient_object_flag = o_star_solar_wind }
				destroy_ambient_object = THIS
			}
		}
		from = {
			set_country_flag = has_built_or_repaired_megastructure
			set_country_flag = has_built_or_repaired_gigastructure	# Triggers Gigastructural Constructs
		}
		every_system_planet = {
			if = {		limit = { has_modifier = terraforming_candidate }				remove_modifier = terraforming_candidate }
			if = {		limit = { giga_is_barren = yes }								change_pc = pc_barren_cold }
			else_if = {	limit = { OR = { giga_is_molten = yes giga_is_toxic = yes } }	change_pc = pc_frozen }
		}
	}
}
