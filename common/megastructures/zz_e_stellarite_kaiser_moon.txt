stellarite_kaiser_moon_2 = {
	entity = ""

	portrait = "GFX_megastructure_moon"
	upgrade_from = { stellarite_kaiser_moon_1 }

	build_time = 3600
	resources = {
		category = giga_megastructures
		cost = {
			alloys = 30000
			energy = 75000
		}
		upkeep = {
			energy = 750
			alloys = 250
		}
	}

	possible = { from = { has_technology = giga_tech_stellarite_kaiser_moon } }

	ai_weight = { factor = value:giga_ai_base_continue } # If we can build it, we will build it
	on_build_complete = {
		save_event_target_as = giga_system
		if = { limit = { exists = fromfrom.planet } fromfrom.planet = { save_event_target_as = giga_planet } }
		from = {
			save_event_target_as = kaiser_moon_builder
			country_event = { id = giga_katzen.0907 }				# Notification
			set_country_flag = has_built_or_repaired_megastructure	# Triggers Galactic Wonders
			create_fleet = {
				name = "Stellarite Kaiser Moon"
				settings = {
					spawn_debris = no
					is_boss = yes
				}
				effect = {
					set_owner = event_target:kaiser_moon_builder
					create_ship = {
						name = "Stellarite Kaiser Moon"
						design = "Stellarite Kaiser Moon"
						upgradable = no
						effect = {
							add_modifier = {
								modifier = stellarite_kaiser_moon_modifier
								days = -1
							}
						}
					}
					set_location = event_target:giga_planet
				}
			}
		}
		fromfrom.planet = { remove_planet = yes }
		remove_megastructure = fromfrom
	}
