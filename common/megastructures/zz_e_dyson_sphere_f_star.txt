#Dyson Sphere
dyson_sphere_5_f_star = {
	entity = "dyson_sphere_phase_05_entity"
	construction_entity = "dyson_sphere_part_4_entity"
	portrait = "GFX_megastructure_dyson_sphere_background"
	upgrade_from = { dyson_sphere_4_f_star }
	prerequisites = { "tech_dyson_sphere" }
	ai_weight = { factor = value:giga_ai_base_continue } # If we can build it, we will build it

	build_time = @dyson_p5_buildtime_f_star
	resources = {
		category = giga_megastructures
		cost = { alloys = @dyson_p5_cost_alloys_f_star unity = @dyson_cost_unity_f_star }
		cost = { alloys = -1 multiplier = value:giga_ai_savings_cost|CATEGORY|important|RESOURCE|alloys|AMOUNT|@dyson_p5_cost_alloys_f_star| }
		cost = { unity = -1 multiplier = value:giga_ai_savings_cost|CATEGORY|important|RESOURCE|unity|AMOUNT|@dyson_cost_unity_f_star| }
		produces = { energy = @dyson_p5_produces_energy_f_star }
	}

	on_build_start = {
		giga_ai_savings_withdraw = { CATEGORY = important RESOURCE = alloys AMOUNT = @dyson_p5_cost_alloys_f_star }
		giga_ai_savings_withdraw = { CATEGORY = important RESOURCE = unity  AMOUNT = @dyson_cost_unity_f_star }
	}

	on_build_complete = {
		if = {
			limit = {
				count_system_planet = {
					limit = { is_star = yes }
					count = 1
				}
			}
			set_star_class = sc_dyson_star
			every_system_planet = {
				limit = { is_star = yes }
				change_pc = pc_f_star
			}
		}
		from = {
			set_country_flag = has_built_or_repaired_megastructure
			set_country_flag = has_built_or_repaired_gigastructure	# Triggers Gigastructural Constructs
		}
		every_system_planet = {
			if = {		limit = { has_modifier = terraforming_candidate }				remove_modifier = terraforming_candidate }
			if = {		limit = { giga_is_barren = yes }								change_pc = pc_barren_cold }
			else_if = {	limit = { OR = { giga_is_molten = yes giga_is_toxic = yes } }	change_pc = pc_frozen }
		}
	}
}

dyson_sphere_restored_f_star = {
	entity = "dyson_sphere_phase_05_entity"
	construction_entity = "dyson_sphere_part_4_entity"
	portrait = "GFX_megastructure_dyson_sphere_background"
	upgrade_from = { dyson_sphere_ruined_f_star }

	build_time = @dyson_r_buildtime_f_star
	resources = {
		category = giga_megastructures
		cost = { alloys = @dyson_r_cost_alloys_f_star unity = @dyson_cost_unity_f_star }
		cost = { alloys = -1 multiplier = value:giga_ai_savings_cost|CATEGORY|important|RESOURCE|alloys|AMOUNT|@dyson_r_cost_alloys_f_star| }
		cost = { unity = -1 multiplier = value:giga_ai_savings_cost|CATEGORY|important|RESOURCE|unity|AMOUNT|@dyson_cost_unity_f_star| }
		produces = { energy = @dyson_r_produces_energy_f_star }
	}

	on_build_start = {
		giga_ai_savings_withdraw = { CATEGORY = important RESOURCE = alloys AMOUNT = @dyson_r_cost_alloys_f_star }
		giga_ai_savings_withdraw = { CATEGORY = important RESOURCE = unity  AMOUNT = @dyson_cost_unity_f_star }
		fromfrom = { set_graphical_culture = root.from }
	}

	possible = {
		exists = from
		from = { has_technology = tech_mega_engineering }
	}

	on_build_complete = {
		if = {
			limit = {
				count_system_planet = {
					limit = { is_star = yes }
					count = 1
				}
			}
			set_star_class = sc_dyson_star
		}
		every_system_planet = {
			if = {		limit = { has_modifier = terraforming_candidate }				remove_modifier = terraforming_candidate }
			if = {		limit = { giga_is_barren = yes }								change_pc = pc_barren_cold }
			else_if = {	limit = { OR = { giga_is_molten = yes giga_is_toxic = yes } }	change_pc = pc_frozen }
		}
		giga_remove_system_debris = yes
		from = {
			set_country_flag = has_built_or_repaired_megastructure
			set_country_flag = has_built_or_repaired_gigastructure	# Triggers Gigastructural Constructs
			country_event = { id = utopia.2011 }
		}
		fromfrom.planet = {
			change_pc = pc_f_star
			giga_set_has_mega_flag = yes
		}
	}
}
