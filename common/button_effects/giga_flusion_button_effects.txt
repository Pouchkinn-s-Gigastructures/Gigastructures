#Propaganda: Reduces Katzen morale (stacks up to 5 times), might spawn some armies!
giga_solak_abilities_propaganda = {
	potential = {
		exists = from
		has_global_flag = solak_resistance_active
		has_global_flag = giga_storming_pyramid_active
		from = {
			has_country_flag = contacted_solak_rebels
			has_country_flag = giga_storming_pyramid_country
		}
		event_target:gigaflusion = {
			owner = { exists = THIS is_country_type = katzenartig_imperium }
			has_ground_combat = yes
		}
	}
	allow = {
		from = {
			resource_stockpile_compare = {
				resource = influence
				value >= 50
			}
		}
		custom_tooltip = {
			fail_text = "cant_if_solak_ability_cooldown"
			hidden_trigger = { NOT = { has_global_flag = giga_storming_pyramid_ability_cooldown } }
		}
	}
	effect = {
		custom_tooltip = "giga_solak_ability_propaganda_tooltip"
		from = {
			add_resource = {
				influence = -50
			}
		}
		event_target:flusion_primitives_country = {
			add_modifier = {
				modifier = solakian_ability_morale_debuff
				days = 180
				multiplier = event_target:flusion_primitives_country.solakian_ability_morale_used
			}
		}
		hidden_effect = {
			event_target:flusion_primitives_country = {
				if = {
					limit = {
						check_variable = {
							which = solakian_ability_morale_used
							value < 5
						}
					}
					change_variable = {
						which = solakian_ability_morale_used
						value = 1
					}
				}
				if = {
					limit = { has_modifier = solakian_ability_morale_debuff }
					remove_modifier = solakian_ability_morale_debuff
				}
			}
			from = { country_event = { id = giga_flusionoperations.6012 } }
		}
	}
}

giga_contact_solak = {
	potential = {
		exists = from
		from = {
			has_country_flag = gigas_diplo_engaged
			NOT = { has_country_flag = giga_diplo_sa_buttons }
			OR = {
				AND = { 
					has_global_flag = solak_resistance_active
					has_country_flag = contacted_solak_rebels
					NOR = {
						has_global_flag = katzenartig_imperium_defeated
						has_country_flag = katzen_vassal
					}
					any_country = { is_country_type = katzenartig_imperium }
				}
				check_variable = { which = ehof_phase value >= 1 } 
			}
		}
	}
	allow = {
		custom_tooltip = {
			fail_text = "giga_not_contacted"
			from = {
				has_global_flag = solak_resistance_active
				has_country_flag = contacted_solak_rebels
			}
		}
		custom_tooltip = {
			fail_text = "giga_katzens_defeated"
			NOT = { has_global_flag = katzenartig_imperium_defeated }
		}
		custom_tooltip = {
			fail_text = "giga_katzens_vassal_solak"
			from = { NOT = { has_country_flag = katzen_vassal } }
		}
		custom_tooltip = { 
			fail_text = "giga_solak_open"
			from = { NOT = { has_country_flag = flusion_map_active } }
		}
	}
	effect = { custom_tooltip = "giga_contact_solak_tooltip" hidden_effect = { from = { country_event = { id = giga_katrebels.2002 } } } }
}
